create table payment (
   pay_id	char(7)   	not null,
   pay_date    	date		not null,
   pay_time    	number(4) 	not null,
   deposit	number(7,2)     not null,
   rental_amt	number(7,2)	not null,
   insurance_amt number(7,2)    not null,
   pay_amt       number(7,2)    not null,
   pay_stat	varchar(10)     not null,
   receipt_no	char(7)         not null,
   reserv_id	char(7)	        not null,
primary key(pay_id),
foreign key(reserv_id) references reservation(reserv_id)
);
