create table reservation (
   	reserv_id	  char(7) 	not null,
   	start_date	  date 	     	not null,
   	rental_duration   number(2)     not null,
   	reservation_stat  varchar(10)   not null,
   	pick_location     varchar(5)    not null,
   	drop_location     varchar(5)	not null,
   	cust_id	          char(7)   	not null,
   	car_id	          char(5)   	not null,
primary key(reserv_id),
foreign key(cust_id) references customer(cust_id),
foreign key(car_id) references car(car_id)
);
