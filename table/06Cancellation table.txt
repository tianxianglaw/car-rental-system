create table cancellation (
   cancel_id		char(7)   	not null,
   cancel_date		date		not null,
   cancel_reason   	varchar(50)	not null,
   refund_amt    	number(7,2)	not null,
   penalty_fee    	number(7,2) 	not null,
   reserv_id		char(7)	        not null,
primary key(cancel_id),
foreign key(reserv_id) references reservation(reserv_id)
);
