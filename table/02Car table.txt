create table car (
	car_id	    char(5)	 not null,
	brand	    varchar(8)	 not null,
	model	    varchar(8)	 not null,
	manu_year   number(4)	 not null,
	plate_no    varchar(10)   not null,
	car_seat    number(1)	 not null,
	car_status  varchar(10)  not null,
	car_colour  varchar(10)  not null,
	trans_type  varchar(20)   not null,
primary key(car_id)
);
